def createSeq(dataset, look_back, foresight):
    X, Y = [], []
    for i in range(len(dataset)-look_back-foresight):
        obs = dataset[i:(i + look_back), 0]
        X.append(obs)
        Y.append(dataset[i + (look_back + foresight), 0])
    return np.array(X), np.array(Y)
